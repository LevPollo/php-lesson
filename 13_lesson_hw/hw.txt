-- Выбрать все телефоны которые есть в наличии, отсортировать по порядку алфавита
SELECT name FROM products ORDER BY name ASC;

-- Сгрупировать все телефоны по моделям
SELECT * FROM `products` GROUP BY manufacturer;

-- Вывести все телефоны с моделью iPhone дороже 2000 рублей
SELECT * FROM `products` WHERE manufacturer = 'iPhone' AND (price > 2000 AND currency = 'BYN') 
OR (price > 746.06 AND currency = 'EUR') OR (price>800 AND currency = 'USD');

-- Найти телефоны только линейки plus либо max
SELECT name FROM `products` WHERE name LIKE '%plus%' OR name LIKE '%max%';

-- Найти телефон в наличии и ценовом диапазане от 1000 до 2000 р
SELECT * FROM `products` WHERE count > 0 AND (price BETWEEN 1000 AND 2000 AND currency = 'BYN') 
OR (price BETWEEN 373 AND 746 AND currency = 'EUR') OR (price BETWEEN 400 AND 800 AND currency = 'USD';

-- Найти все телефоны моделей Google, iPhone, Samsung
SELECT manufacturer FROM `products` WHERE manufacturer = 'Samsung' 
OR manufacturer = 'iPhone'
OR manufacturer = 'Google';

-- Сделать запрос который выведет среднюю цену всех телефонов и количество таких телефонов сгруппированнах по моделям (использовать alias)
средняя цена каждой модели телефона, количество каждой модели телефона, модель телефона 
SELECT manufacturer, COUNT(manufacturer) AS manufacturer_count, sum(count),
AVG(CASE
    WHEN currency = 'USD' THEN price * 2.4
     WHEN currency = 'EUR' THEN price * 3
    ELSE price
END) AS average_number_manufacturer
FROM `products`
GROUP BY manufacturer;
// модель/количество линеек/количество всех телефонов/средняя цена модели
